   ```typescript
   import { renderHook } from '@testing-library/react'
   import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'
   import { useWebSocket } from '../hooks/useWebSocket'

   // Mock WebSocket globally
   const mockWs = {
     close: vi.fn(),
     addEventListener: vi.fn(),
     removeEventListener: vi.fn(),
     readyState: 0,
   }

   beforeEach(() => {
     vi.stubGlobal('WebSocket', vi.fn(() => mockWs))
   })

   afterEach(() => {
     vi.unstubAllGlobals()
   })

   describe('useWebSocket', () => {
     it('returns initial disconnected state when analysisId is null', () => {
       const { result } = renderHook(() => useWebSocket(null))
       expect(result.current.connected).toBe(false)
       expect(result.current.progress).toBe(0)
       expect(result.current.complete).toBe(false)
       expect(result.current.error).toBeNull()
     })

     it('does not create WebSocket when analysisId is null', () => {
       renderHook(() => useWebSocket(null))
       expect(WebSocket).not.toHaveBeenCalled()
     })

     it('creates WebSocket connection when analysisId is provided', () => {
       renderHook(() => useWebSocket('test-id-123'))
       expect(WebSocket).toHaveBeenCalled()
     })
   })
   ```

2. Create frontend/src/test/api-local.test.ts:
   ```typescript
   import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest'

   // Mock fetch before importing the module under test
   const mockFetch = vi.fn()
   beforeEach(() => {
     vi.stubGlobal('fetch', mockFetch)
     mockFetch.mockResolvedValue({
       ok: true,
       json: async () => [],
       text: async () => '[]',
     })
   })

   afterEach(() => {
     vi.unstubAllGlobals()
     vi.clearAllMocks()
   })

   describe('analysisApi', () => {
     it('list() calls GET /api/analysis', async () => {
       const { analysisApi } = await import('../lib/api-local')
       await analysisApi.list()
       expect(mockFetch).toHaveBeenCalledWith(
         expect.stringContaining('/api/analysis'),
         expect.objectContaining({ method: expect.any(String) })
       )
     })

     it('get(id) calls GET /api/analysis/{id}', async () => {
       const { analysisApi } = await import('../lib/api-local')
       await analysisApi.get('42')
       expect(mockFetch).toHaveBeenCalledWith(
         expect.stringContaining('/api/analysis/42'),
         expect.anything()
       )
     })

     it('create() calls POST /api/analysis', async () => {
       mockFetch.mockResolvedValueOnce({
         ok: true,
         json: async () => ({ id: 'new-id' }),
       })
       const { analysisApi } = await import('../lib/api-local')
       const result = await analysisApi.create({ videoId: 'v1', mode: 'all' })
       expect(result.id).toBe('new-id')
       const [url, opts] = mockFetch.mock.calls[0]
       expect(url).toContain('/api/analysis')
       expect(opts.method).toBe('POST')
     })
   })

   describe('videoApi', () => {
     it('list() calls GET /api/videos', async () => {
       const { videoApi } = await import('../lib/api-local')
       await videoApi.list()
       expect(mockFetch).toHaveBeenCalledWith(
         expect.stringContaining('/api/videos'),
         expect.anything()
       )
     })
   })
   ```
